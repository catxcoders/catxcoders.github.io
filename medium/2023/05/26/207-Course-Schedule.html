class Solution(object):
    def canFinish(self, numCourses, prerequisites):
        """
        :type numCourses: int
        :type prerequisites: List[List[int]]
        :rtype: bool
        """
        def dfs(i):
            if v[i] == -1:
                return False
            if v[i] == 1:
                return True
            v[i] = -1 # visiting
            for j in graph[i]:
                if not dfs(j):
                    return False
            v[i] = 1 # visited
            return True

        graph = {}
        for i in range(numCourses):
            graph[i] = []
        for pre in prerequisites:
            graph[pre[1]].append(pre[0])

        v = [0] * numCourses
        for i in range(numCourses):
            if not dfs(i):
                return False
        return True
