class Solution:
    def quickSelect(self, nums, k):
        # éپ‍è؟´ه¯¦çڈ¾ Quick Select ç®—و³•
        if not nums:
            return None

        # éپ¸و“‡ن¸€ه€‹éڑ¨و©ںهں؛و؛–ه…ƒç´ 
        pivot = random.choice(nums)

        # هˆ‌ه§‹هŒ–2ه€‹هˆ—è،¨ï¼Œهˆ†هˆ¥ه­کو”¾ه¤§و–¼م€په°ڈو–¼pivot
        L = [x for x in nums if x > pivot]
        R = [x for x in nums if x < pivot]
        
        # ç­‰و–¼هں؛و؛–ه…ƒç´ çڑ„ه…ƒç´ ه€‹و•¸
        M_len = len(nums)-len(L)-len(R)

        # éپ‍è؟´هں·è،Œ Quick Select
        if k <= len(L):
            return self.quickSelect(L, k)
        elif k <= len(L) + M_len:
            return pivot
        else:
            return self.quickSelect(R, k - len(L) - M_len)

    def findKthLargest(self, nums: List[int], k: int) -> int:
        return self.quickSelect(nums, k)
