class Solution:
    def minCapability(self, nums: List[int], k: int) -> int:
        min_reward, max_reward = min(nums), max(nums)

        while min_reward < max_reward:
            mid = (min_reward + max_reward) // 2
            possible_thefts = 0

            i = 0
            while i < len(nums):
                if nums[i] <= mid:
                    possible_thefts += 1
                    i += 2
                else:
                    i += 1
            if possible_thefts >= k:
                max_reward = mid
            else:
                min_reward = mid + 1
        return min_reward
