class Solution(object):
    def numberOfSubarrays(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: int
        """
        ht = {0: 1}
        
        num_nice_arr = 0
        cur = 0
        for num in nums:
            if num % 2 == 1:
                cur += 1
            ht.setdefault(cur, 0)
            ht[cur] += 1
            if cur - k in ht:
                num_nice_arr += ht[cur-k]
        return num_nice_arr
