1438. Longest Continuous Subarray With Absolute Diff Less Than or Equal to Limit зөҰе®ҡдёҖеҖӢж•ҙж•ёж•ёзө„ nums е’ҢдёҖеҖӢж•ҙж•ё limitпјҢи«Ӣиҝ”еӣһй•·еәҰжңҖеӨ§зҡ„йқһз©әеӯҗйҷЈеҲ—зҡ„й•·еәҰпјҢдҪҝеҫ—и©ІеӯҗйҷЈеҲ—дёӯд»»дҪ•е…©еҖӢе…ғзҙ д№Ӣй–“зҡ„зө•е°Қе·®дёҚи¶…йҒҺ limitгҖӮ

и§ЈйЎҢжҖқи·Ҝ

йҖҷйҒ“йЎҢзӣ®иҰҒжұӮжҲ‘еҖ‘жүҫеҮәдёҖеҖӢйҖЈзәҢеӯҗйҷЈеҲ—пјҢдҪҝеҫ—и©ІеӯҗйҷЈеҲ—дёӯзҡ„жңҖеӨ§еҖјиҲҮжңҖе°ҸеҖјд№Ӣй–“зҡ„е·®дёҚи¶…йҒҺ limitгҖӮжҲ‘еҖ‘еҸҜд»Ҙз”ЁйӣҷжҢҮйҮқе’Ңе…©еҖӢе–®иӘҝйҒһеўһеҸҠе–®иӘҝйҒһжёӣзҡ„йӣҷз«ҜйҡҠеҲ—(deque)дҫҶи§ЈжұәйҖҷеҖӢе•ҸйЎҢгҖӮ

е…·й«”жҖқи·ҜеҰӮдёӢпјҡ

  1. йӣҷз«ҜйҡҠеҲ—(deque)пјҡ
    • дҪҝз”Ё increaseDeque дҫҶз¶ӯиӯ·дёҖеҖӢе–®иӘҝйҒһеўһзҡ„йҡҠеҲ—пјҢйҡҠеҲ—дёӯзҡ„е…ғзҙ еңЁindexиҲҮж•ёеҖјдёҠйғҪжҳҜйҒһеўһзҡ„пјҢе°ҚжҮүзҡ„жҳҜз•¶еүҚеӯҗйҷЈеҲ—дёӯзҡ„жңҖе°ҸеҖјгҖӮ
    • дҪҝз”Ё decreaseDeque дҫҶз¶ӯиӯ·дёҖеҖӢе–®иӘҝйҒһжёӣзҡ„йҡҠеҲ—пјҢйҡҠеҲ—дёӯзҡ„е…ғзҙ еңЁindexдёҠжҳҜйҒһеўһзҡ„пјҢеңЁж•ёеҖјдёҠжҳҜйҒһжёӣзҡ„пјҢйҡҠеҲ—дёӯзҡ„е…ғзҙ е°ҚжҮүзҡ„жҳҜз•¶еүҚеӯҗйҷЈеҲ—дёӯзҡ„жңҖеӨ§еҖјгҖӮ

      жҲ‘еҖ‘еңЁеҜҰдҪңзҡ„жҷӮеҖҷжҳҜеӯҳindexеңЁйӣҷз«ҜйҡҠеҲ—(deque)дёӯпјҢж–№дҫҝиЁҲз®—й•·еәҰгҖӮ

  2. йҒҚжӯ·йҷЈеҲ—пјҡ
    • е°Қж–јйҷЈеҲ—дёӯзҡ„жҜҸеҖӢе…ғзҙ пјҢе°Үе…¶еҠ е…Ҙ increaseDeque е’Ң decreaseDeque дёӯпјҢдёҰдҝқжҢҒе…©еҖӢйҡҠеҲ—зҡ„е–®иӘҝжҖ§гҖӮ
    • еңЁжҜҸж¬ЎеҠ е…Ҙж–°е…ғзҙ еҫҢпјҢжӘўжҹҘз•¶еүҚеӯҗйҷЈеҲ—зҡ„жңҖеӨ§еҖјиҲҮжңҖе°ҸеҖјд№Ӣй–“зҡ„е·®жҳҜеҗҰи¶…йҒҺдәҶ limitгҖӮ
    • еҰӮжһңи¶…йҒҺдәҶ limitпјҢжҲ‘еҖ‘йңҖиҰҒиӘҝж•ҙе·ҰйӮҠз•Ң lпјҢзӣҙеҲ°еӯҗйҷЈеҲ—еҶҚж¬Ўз¬ҰеҗҲжўқд»¶гҖӮ еҰӮжһңзҸҫеңЁеҠ е…Ҙзҡ„е…ғзҙ жңғе°ҺиҮҙжңҖеӨ§еҖјиҲҮжңҖе°ҸеҖјд№Ӣй–“зҡ„е·®жҳҜеҗҰи¶…йҒҺдәҶ limitпјҢйӮЈеҸӘжңүе…©зЁ®жғ…жіҒ:
      1. еҠ е…Ҙзҡ„еҖјзӮәеҚҖй–“зҡ„жңҖеӨ§еҖјпјҢжҲ‘еҖ‘иҰҒиӘҝж•ҙе·ҰйӮҠз•ҢlдҪҝеҫ—жңҖе°ҸеҖји®ҠеӨ§еҲ°з¬ҰеҗҲ жңҖеӨ§еҖјиҲҮжңҖе°ҸеҖјд№Ӣй–“зҡ„е·® <= limit пјҢз”ұж–јincreaseDeque йҡҠеҲ—дёӯзҡ„е…ғзҙ еңЁindexиҲҮж•ёеҖјдёҠйғҪжҳҜйҒһеўһзҡ„пјҢеӣ жӯӨжҲ‘еҖ‘еҸҜд»ҘйҖҸйҒҺincreaseDequeжүҫжңҖе°Ҹз¬ҰеҗҲйҷҗеҲ¶зҡ„е·Ұз•ҢгҖӮ
      2. еҠ е…Ҙзҡ„еҖјзӮәеҚҖй–“зҡ„жңҖе°ҸеҖјпјҢиҲҮеҠ е…ҘзӮәжңҖеӨ§еҖјйЎһдјјпјҢеҸӘжҳҜжҸӣжҲҗйҖҸйҒҺdecreaseDequeжүҫжңҖе°Ҹзҡ„indexз¬ҰеҗҲйҷҗеҲ¶зҡ„е·Ұз•ҢгҖӮ
  3. жӣҙж–°зөҗжһңпјҡ
    • еңЁиӘҝж•ҙе·ҰйӮҠз•ҢеҫҢпјҢиЁҲз®—з•¶еүҚеӯҗйҷЈеҲ—зҡ„й•·еәҰпјҢдёҰжӣҙж–°жңҖеӨ§еӯҗйҷЈеҲ—й•·еәҰгҖӮ

йҖҷжЁЈпјҢжҲ‘еҖ‘е°ұеҸҜд»ҘеңЁз·ҡжҖ§жҷӮй–“е…§жүҫеҲ°з¬ҰеҗҲжўқд»¶зҡ„жңҖеӨ§еӯҗйҷЈеҲ—зҡ„й•·еәҰгҖӮ

Python еҜҰдҪң

class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        increaseDeque = deque()  # е–®иӘҝйҒһеўһйҡҠеҲ—
        decreaseDeque = deque()  # е–®иӘҝйҒһжёӣйҡҠеҲ—
        l = 0  # е·ҰйӮҠз•Ң
        max_length = 0  # жңҖй•·зҡ„еӯҗйҷЈеҲ—й•·еәҰ
        
        for r in range(len(nums)):
            # з¶ӯиӯ·е–®иӘҝйҒһеўһйҡҠеҲ—
            while increaseDeque and nums[r] < nums[increaseDeque[-1]]:
                increaseDeque.pop()
            increaseDeque.append(r)
            
            # з¶ӯиӯ·е–®иӘҝйҒһжёӣйҡҠеҲ—
            while decreaseDeque and nums[r] > nums[decreaseDeque[-1]]:
                decreaseDeque.pop()
            decreaseDeque.append(r)
            
            # з•¶еүҚеӯҗйҷЈеҲ—дёҚз¬ҰеҗҲжўқд»¶жҷӮпјҢз§»еӢ•е·ҰйӮҠз•Ң
            while nums[decreaseDeque[0]] - nums[increaseDeque[0]] > limit:
                l += 1
                if increaseDeque[0] < l:
                    increaseDeque.popleft()
                if decreaseDeque[0] < l:
                    decreaseDeque.popleft()
            
            # жӣҙж–°жңҖеӨ§еӯҗйҷЈеҲ—й•·еәҰ
            max_length = max(max_length, r - l + 1)
        
        return max_length

C++ еҜҰдҪң

class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        // maintain two monotonic stacks
        deque<int> increaseSt; // Holds indices in increasing order, front is the minimum of the interval.
        deque<int> decreaseSt; // Holds indices in decreasing order, front is the maximum of the interval.
        int l = -1; // left bound of the interval, (exclusive)
        int r; // right bound of the interval, (inclusive)
        int sizeOfSubArr = 1;

        // traverse each number in nums, treating it as the right end of interval
        for (r = 0; r<nums.size(); r++){
            // add number in increaseSt and decreaseSt, and maintain two monotonic stacks
            while(!increaseSt.empty() && nums[r] < nums[increaseSt.back()]) increaseSt.pop_back();
            while(!decreaseSt.empty() && nums[r] > nums[decreaseSt.back()]) decreaseSt.pop_back();
            increaseSt.push_back(r);
            decreaseSt.push_back(r);
            
            // check the current interval is valid
            if (nums[decreaseSt.front()]-nums[increaseSt.front()]>limit){
                sizeOfSubArr = max(sizeOfSubArr, r-l-1);
                
                // move the left of interval to make the interval valid
                while(nums[decreaseSt.front()]-nums[increaseSt.front()]>limit){
                    if (decreaseSt.front() == r){
                        l = increaseSt.front();
                        increaseSt.pop_front();
                    }else{
                        l = decreaseSt.front();
                        decreaseSt.pop_front();
                    }
                }
            }
        }
        sizeOfSubArr = max(sizeOfSubArr, r-l-1);
        return sizeOfSubArr;
    }
};
