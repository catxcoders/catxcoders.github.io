1110. Delete Nodes And Return Forest з•¶дҪ жңүдёҖжЈөдәҢе…ғжЁ№пјҢе…¶дёӯжҜҸеҖӢзҜҖй»һзҡ„еҖјйғҪдёҚдёҖжЁЈпјҢжҲ‘еҖ‘зҡ„зӣ®жЁҷжҳҜеҲӘйҷӨдёҖдәӣзҜҖй»һпјҢйҖҷдәӣзҜҖй»һзҡ„еҖјеңЁдёҖеҖӢеҸ« to_delete зҡ„еҲ—иЎЁиЈЎгҖӮз•¶йҖҷдәӣзҜҖй»һиў«еҲӘйҷӨеҫҢпјҢжЁ№жңғиў«еҲҶжҲҗе№ҫеҖӢдёҚйҖЈжҺҘзҡ„е°ҸжЁ№пјҢйҖҷдәӣе°ҸжЁ№иў«зЁұзӮәжЈ®жһ—гҖӮдҪ йңҖиҰҒиҝ”еӣһйҖҷдәӣе°ҸжЁ№зҡ„ж №зҜҖй»һпјҢй ҶеәҸжІ’жңүй—ңдҝӮгҖӮ

и§ЈйЎҢжҖқи·Ҝ

йҖҷйЎҢзҡ„йҮҚй»һеңЁж–јпјҢжҲ‘еҖ‘йңҖиҰҒз”ЁйҒһиҝҙзҡ„ж–№ејҸйҒҚжӯ·ж•ҙжЈөжЁ№пјҢйҖҗдёҖжӘўжҹҘжҜҸеҖӢзҜҖй»һжҳҜдёҚжҳҜиҰҒиў«еҲӘжҺүгҖӮеҰӮжһңжҹҗеҖӢзҜҖй»һйңҖиҰҒиў«еҲӘйҷӨпјҢжҲ‘еҖ‘е°ұжҠҠе®ғзҡ„еӯҗзҜҖй»һпјҲеҰӮжһңжңүзҡ„и©ұпјүеҠ е…ҘеҲ°жЈ®жһ—иЈЎпјҢз„¶еҫҢжҠҠйҖҷеҖӢзҜҖй»һеҲӘжҺүгҖӮжңҖзөӮпјҢз•ҷдёӢдҫҶзҡ„зҜҖй»һжңғеҪўжҲҗе№ҫжЈөж–°зҡ„е°ҸжЁ№пјҢйҖҷдәӣе°ҸжЁ№е°ұжҳҜжҲ‘еҖ‘зҡ„гҖҢжЈ®жһ—гҖҚгҖӮ

е…·й«”жӯҘй©ҹпјҡ

  1. дҪҝз”Ёж·ұеәҰе„Әе…Ҳжҗңзҙў (DFS) йҒҚжӯ·ж•ҙжЈөжЁ№пјҢдёҰжӘўжҹҘжҜҸеҖӢзҜҖй»һгҖӮ
  2. еҰӮжһңзҜҖй»һзҡ„еҖјеӯҳеңЁж–ј to_delete еҲ—иЎЁдёӯпјҢе°Үе®ғзҡ„е·ҰеӯҗзҜҖй»һе’ҢеҸіеӯҗзҜҖй»һпјҲеҰӮжһңеӯҳеңЁпјүеҠ е…ҘеҲ°жЈ®жһ—дёӯпјҢз„¶еҫҢеҲӘйҷӨжӯӨзҜҖй»һгҖӮ
  3. йҒһиҝҙзөҗжқҹеҫҢпјҢжӘўжҹҘж №зҜҖй»һпјҢеҰӮжһңж №зҜҖй»һжң¬иә«жңӘиў«еҲӘйҷӨпјҢе°Үе…¶еҠ е…ҘжЈ®жһ—гҖӮ
  4. дҪҝз”ЁдёҖеҖӢhash set toDelete дҫҶеҝ«йҖҹжҹҘжүҫиҰҒеҲӘйҷӨзҡ„зҜҖй»һгҖӮ

Python еҜҰдҪң

# е®ҡзҫ©дәҢе…ғжЁ№зҜҖй»һ
class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

class Solution:
    def delNodes(self, root: TreeNode, to_delete: list[int]) -> list[TreeNode]:
        toDelete = set(to_delete)  # дҪҝз”ЁйӣҶеҗҲдҫҶжҸҗй«ҳжҹҘжүҫж•ҲзҺҮ
        forest = []  # еӯҳж”ҫзөҗжһңзҡ„жЈ®жһ—

        def dfs(node: TreeNode) -> TreeNode:
            if not node:
                return None

            # йҒһиҝҙиҷ•зҗҶе·ҰеӯҗжЁ№е’ҢеҸіеӯҗжЁ№
            node.left = dfs(node.left)
            node.right = dfs(node.right)

            # еҰӮжһңз•¶еүҚзҜҖй»һйңҖиҰҒиў«еҲӘйҷӨ
            if node.val in toDelete:
                if node.left:
                    forest.append(node.left)  # е·ҰеӯҗжЁ№еҠ е…ҘжЈ®жһ—
                if node.right:
                    forest.append(node.right)  # еҸіеӯҗжЁ№еҠ е…ҘжЈ®жһ—
                return None  # еҲӘйҷӨз•¶еүҚзҜҖй»һпјҢиҝ”еӣһ None

            return node  # дҝқз•ҷз•¶еүҚзҜҖй»һ

        # иҷ•зҗҶж №зҜҖй»һ
        if dfs(root):
            forest.append(root)

        return forest

C++ еҜҰдҪң

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<TreeNode*> forest;
    void dfs(TreeNode* &root, const unordered_set<int> &toDelete){
        if (!root) return;
        dfs(root -> left, toDelete);
        dfs(root -> right, toDelete);
        if (toDelete.find(root->val)!=toDelete.end()){
            if (root -> left) forest.push_back(root->left);
            if (root -> right) forest.push_back(root->right);
            delete root;
            root = nullptr;
        }
        return;
    }

    vector<TreeNode*> delNodes(TreeNode* root, vector<int>& to_delete) {
        
        unordered_set<int> toDelete(to_delete.begin(), to_delete.end());

        dfs(root, toDelete);
        if (root) forest.push_back(root);
        
        return forest;

    }
};
