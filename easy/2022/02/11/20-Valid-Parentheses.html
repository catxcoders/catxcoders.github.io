class Solution:
    def isValid(self, s: str) -> bool:
        stack = []
        for c in s:
            if c in ['(', '[', '{']:
                stack.append(c)
            else:
                if not stack:
                    return False
                target = stack.pop()
                if target == '(' and c != ')':
                    return False
                if target == "[" and c != ']':
                    return False
                if target == '{' and c != '}':
                    return False
        if not stack:
            return True
        return False
