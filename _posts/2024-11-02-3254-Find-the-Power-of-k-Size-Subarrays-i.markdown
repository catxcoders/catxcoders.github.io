```python
class Solution:
    def resultsArray(self, nums: List[int], k: int) -> List[int]:
        if k == 1:
            return nums
        m = 1
        ans = []
        for i in range(1, len(nums)):
            if nums[i] == nums[i-1]+1:
                m += 1
            else:
                m = 1
            if i >= k-1:
                if m == k:
                    ans.append(nums[i])
                    m -= 1
                else:
                    ans.append(-1)
        return ans
```
